## 执行步骤

```bash
cd /tmp  # 在临时目录跑，不污染主项目

# 1. 创建新项目
PYTHONPATH=/Users/popmart/Documents/python/forge-agent/src \
  python -m forge_agent.cli new my_app --template=basic

cd my_app

# 2. 安装依赖
pip install -e /Users/popmart/Documents/python/forge-agent

# 3. 查看 LLM providers
PYTHONPATH=/Users/popmart/Documents/python/forge-agent/src \
  forge-agent llm list
# 期望: 看到 mock, openai, anthropic, gemini, ollama 5 个 provider

# 4. 生成一个简单 agent (用 mock provider)
PYTHONPATH=/Users/popmart/Documents/python/forge-agent/src \
  forge-agent generate "echo agent that returns hello world" \
  --provider=mock

# 5. 查看生成的 agents
PYTHONPATH=/Users/popmart/Documents/python/forge-agent/src \
  forge-agent list
# 期望: 看到 1 个 agent

# 6. 写一个测试脚本运行 agent
cat > test_run.py <<'PYEOF'
import asyncio
from forge_agent import AgentContext
from agents.example import ExampleAgent

async def main():
    agent = ExampleAgent()
    await agent.initialize()
    report = await agent.run(AgentContext(scope_id="t1", scope_name="test"))
    print(f"Verdict: {report.verdict}")
    print(f"Evidence: {report.evidence}")
    print(f"Run ID: {report.run_id}")
    await agent.shutdown()
    assert report.verdict is not None, "verdict is None"
    assert report.run_id, "run_id is empty"

asyncio.run(main())
PYEOF

# 7. 跑测试
python test_run.py
# 期望: 看到 "Verdict: ..." 等输出，无异常
```
