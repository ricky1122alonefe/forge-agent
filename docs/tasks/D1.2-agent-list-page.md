### Step 3: 更新 `routes/pages.py`

```python
# src/forge_agent/dashboard/routes/pages.py
"""Page routes (HTML)."""
from __future__ import annotations

from fastapi import APIRouter, Request
from fastapi.responses import HTMLResponse
from fastapi.templating import Jinja2Templates

from forge_agent.registry.registry import get_registry

router = APIRouter()
templates = Jinja2Templates(directory="src/forge_agent/dashboard/templates")


@router.get("/", response_class=HTMLResponse)
async def index(request: Request) -> HTMLResponse:
    """Home: list all registered agents."""
    registry = get_registry()
    agents = []
    for agent_id in registry.list_agents():
        cls = registry.get_class(agent_id)
        meta = registry.get_metadata(agent_id) or {}
        agents.append({
            "agent_id": agent_id,
            "name": getattr(cls, "name", agent_id),
            "domain": getattr(cls, "domain", "generic"),
            "version": getattr(cls, "version", "0.0.0"),
            "capabilities": meta.get("capabilities", []),
        })
    return templates.TemplateResponse(
        "index.html",
        {"request": request, "agents": agents},
    )
```

```python
# tests/unit/dashboard/test_pages.py
"""Tests for page routes."""
from __future__ import annotations

from pathlib import Path

import pytest
from fastapi.testclient import TestClient

from forge_agent.core.base import BaseAgent
from forge_agent.core.contracts import AgentReport
from forge_agent.core.context import AgentContext
from forge_agent.core.enums import Verdict
from forge_agent.dashboard.app import create_app
from forge_agent.registry.decorators import register_agent


@register_agent(domain="test", tags=["example"])
class _Demo(BaseAgent):
    agent_id = "test.demo"
    name = "Demo Agent"
    version = "0.1.0"
    domain = "test"

    async def observe(self, ctx): return {}
    async def decide(self, ctx, obs): return {}
    async def act(self, ctx, dec):
        return AgentReport(agent_id=self.agent_id, name=self.name, verdict=Verdict.NEUTRAL)


@pytest.fixture
def client(tmp_path: Path) -> TestClient:
    return TestClient(create_app(project_root=tmp_path))


def test_index_renders(client: TestClient) -> None:
    r = client.get("/")
    assert r.status_code == 200
    assert "forge-agent Dashboard" in r.text


def test_index_lists_agents(client: TestClient) -> None:
    r = client.get("/")
    assert r.status_code == 200
    assert "test.demo" in r.text
    assert "Demo Agent" in r.text
    assert "test" in r.text


def test_index_handles_no_agents(client: TestClient) -> None:
    # If no agents registered, should show empty state
    from forge_agent.registry.registry import get_registry
    get_registry()._classes.clear()
    r = client.get("/")
    assert r.status_code == 200
    assert "No agents registered" in r.text or "test.demo" in r.text  # either is fine
```
