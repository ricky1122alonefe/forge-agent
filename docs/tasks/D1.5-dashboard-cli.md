```python
# src/forge_agent/cli/cmd_dashboard.py
"""`forge-agent dashboard` — start the local observability dashboard."""
from __future__ import annotations

import argparse
import sys


def add(sub: argparse._SubParsersAction) -> None:
    p = sub.add_parser(
        "dashboard",
        help="Start the local observability dashboard (web UI)",
    )
    p.add_argument(
        "--host",
        default="127.0.0.1",
        help="Bind host (default: 127.0.0.1, local only)",
    )
    p.add_argument(
        "--port", "-p",
        type=int,
        default=8765,
        help="Bind port (default: 8765)",
    )
    p.add_argument(
        "--reload",
        action="store_true",
        help="Enable hot reload (development only)",
    )
    p.add_argument(
        "--no-browser",
        action="store_true",
        help="Don't auto-open browser",
    )
    p.set_defaults(func=run)


def run(args: argparse.Namespace) -> int:
    """Start the dashboard server."""
    try:
        import uvicorn
    except ImportError:
        print(
            "Error: uvicorn not installed.\n"
            "Install with: pip install 'forge-agent[dashboard]'",
            file=sys.stderr,
        )
        return 1

    from forge_agent.dashboard.app import create_app

    project_root = args.project.resolve()
    app = create_app(
        project_root=project_root,
        host=args.host,
        port=args.port,
    )

    print("⚡ forge-agent Dashboard")
    print(f"  → URL:       http://{args.host}:{args.port}")
    print(f"  → Project:   {project_root}")
    print(f"  → Hot reload: {'ON' if args.reload else 'OFF'}")
    print("  → Press Ctrl+C to stop")
    print()

    # Auto-open browser (best-effort)
    if not args.no_browser:
        import threading
        import time
        import webbrowser

        def _open_browser() -> None:
            time.sleep(1.0)  # wait for server to start
            url = f"http://{args.host}:{args.port}"
            try:
                webbrowser.open(url)
            except Exception:  # noqa: BLE001
                pass

        threading.Thread(target=_open_browser, daemon=True).start()

    uvicorn.run(
        app,
        host=args.host,
        port=args.port,
        reload=args.reload,
        log_level="info",
    )
    return 0
```

### Step 3: 写集成测试

```python
# tests/integration/test_dashboard_cli.py
"""Integration tests for `forge-agent dashboard` CLI."""
from __future__ import annotations

import socket
import subprocess
import sys
import time
from pathlib import Path

import pytest


def _is_port_open(host: str, port: int) -> bool:
    """Check if a TCP port is open."""
    try:
        with socket.create_connection((host, port), timeout=0.5):
            return True
    except (ConnectionRefusedError, socket.timeout, OSError):
        return False


@pytest.mark.integration
def test_dashboard_help() -> None:
    """`forge-agent dashboard --help` exits 0 and shows usage."""
    result = subprocess.run(
        [sys.executable, "-m", "forge_agent.cli", "dashboard", "--help"],
        capture_output=True,
        text=True,
        timeout=10,
        env={"PYTHONPATH": "src", "PATH": "/usr/bin:/bin:/usr/local/bin"},
    )
    assert result.returncode == 0
    assert "Start the local observability dashboard" in result.stdout
    assert "--port" in result.stdout


@pytest.mark.integration
def test_dashboard_starts_and_serves() -> None:
    """Start dashboard on a free port, verify it serves."""
    import uvicorn

    from forge_agent.dashboard.app import create_app

    # Find a free port
    with socket.socket() as s:
        s.bind(("127.0.0.1", 0))
        port = s.getsockname()[1]

    project_root = Path("/tmp")
    app = create_app(project_root=project_root, host="127.0.0.1", port=port)

    config = uvicorn.Config(app, host="127.0.0.1", port=port, log_level="error")
    server = uvicorn.Server(config)

    import threading
    thread = threading.Thread(target=server.run, daemon=True)
    thread.start()

    # Wait for server to start
    for _ in range(50):
        if _is_port_open("127.0.0.1", port):
            break
        time.sleep(0.1)

    assert _is_port_open("127.0.0.1", port), "Dashboard did not start"

    # Make a request
    import urllib.request
    with urllib.request.urlopen(f"http://127.0.0.1:{port}/api/health", timeout=2) as r:
        assert r.status == 200
        assert b'"status":"ok"' in r.read()

    server.should_exit = True
    thread.join(timeout=2)
```
