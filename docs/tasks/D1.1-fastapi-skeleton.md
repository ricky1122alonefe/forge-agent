### Step 2: 创建 `app.py`

```python
# src/forge_agent/dashboard/app.py
"""FastAPI app factory for the forge-agent dashboard."""
from __future__ import annotations

from pathlib import Path
from typing import Any

from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles

from forge_agent.dashboard.routes import api, pages


def create_app(
    *,
    project_root: Path,
    host: str = "127.0.0.1",
    port: int = 8765,
) -> FastAPI:
    """Create a FastAPI app bound to a forge-agent project.

    Args:
        project_root: Directory containing `generated_agents/`.
        host: Bind host (default 127.0.0.1 — local only).
        port: Bind port (default 8765).

    Returns:
        Configured FastAPI app, ready to be served by uvicorn.
    """
    app = FastAPI(
        title="forge-agent Dashboard",
        version="0.3.0",
        description="Local observability panel for forge-agent",
    )

    # Stash project root + host/port in app.state for downstream handlers
    app.state.project_root = Path(project_root).resolve()
    app.state.host = host
    app.state.port = port

    # Mount static files (HTMX / CSS)
    static_dir = Path(__file__).parent / "static"
    app.mount("/static", StaticFiles(directory=static_dir), name="static")

    # Mount routes
    app.include_router(pages.router)
    app.include_router(api.router, prefix="/api")

    return app
```

```python
# tests/unit/dashboard/__init__.py
# tests/unit/dashboard/test_app.py
"""Tests for the dashboard app factory."""
from __future__ import annotations

from pathlib import Path

import pytest
from fastapi.testclient import TestClient

from forge_agent.dashboard.app import create_app


@pytest.fixture
def client(tmp_path: Path) -> TestClient:
    app = create_app(project_root=tmp_path)
    return TestClient(app)


def test_health_endpoint(client: TestClient) -> None:
    r = client.get("/api/health")
    assert r.status_code == 200
    assert r.json() == {"status": "ok"}


def test_index_page(client: TestClient) -> None:
    r = client.get("/")
    assert r.status_code == 200
    assert "forge-agent Dashboard" in r.text


def test_project_root_stored(client: TestClient, tmp_path: Path) -> None:
    assert client.app.state.project_root == tmp_path.resolve()


def test_static_files_served(client: TestClient) -> None:
    r = client.get("/static/")
    # 404 is OK — directory listing not allowed; just verify mount exists
    assert r.status_code in (200, 404)
```
