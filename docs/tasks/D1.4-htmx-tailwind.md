### Step 1: 创建 `static/forge.css`

```css
/* src/forge_agent/dashboard/static/forge.css */
/* Custom theme overrides for forge-agent */

:root {
  --color-primary: #4f46e5;       /* indigo-600 */
  --color-primary-light: #eef2ff;  /* indigo-50 */
  --color-success: #10b981;        /* emerald-500 */
  --color-warning: #f59e0b;        /* amber-500 */
  --color-danger: #ef4444;         /* red-500 */
  --color-bg: #f8fafc;             /* slate-50 */
}

body {
  font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto,
               "Helvetica Neue", Arial, sans-serif;
  -webkit-font-smoothing: antialiased;
}

/* Status pills */
.status-pill {
  display: inline-flex;
  align-items: center;
  padding: 0.125rem 0.5rem;
  border-radius: 9999px;
  font-size: 0.75rem;
  font-weight: 500;
}

.status-pill::before {
  content: "";
  display: inline-block;
  width: 0.5rem;
  height: 0.5rem;
  border-radius: 9999px;
  margin-right: 0.375rem;
}

.status-pill.active::before { background: var(--color-success); }
.status-pill.testing::before { background: var(--color-warning); }
.status-pill.archived::before { background: #94a3b8; }
.status-pill.broken::before { background: var(--color-danger); }

.status-pill.active    { background: #d1fae5; color: #065f46; }
.status-pill.testing   { background: #fef3c7; color: #92400e; }
.status-pill.archived  { background: #f1f5f9; color: #475569; }
.status-pill.broken    { background: #fee2e2; color: #991b1b; }

/* Code blocks */
code, pre, .font-mono {
  font-family: ui-monospace, SFMono-Regular, "SF Mono", Menlo, Consolas, monospace;
}

/* Smooth transitions for HTMX swaps */
.htmx-swapping {
  opacity: 0;
  transition: opacity 200ms ease-out;
}

.htmx-settling {
  opacity: 1;
  transition: opacity 200ms ease-in;
}

/* Loading indicator */
.htmx-indicator {
  display: none;
}
.htmx-request .htmx-indicator {
  display: inline-block;
}
.htmx-request.htmx-indicator {
  display: inline-block;
}

/* Custom scrollbar */
::-webkit-scrollbar { width: 8px; height: 8px; }
::-webkit-scrollbar-track { background: transparent; }
::-webkit-scrollbar-thumb { background: #cbd5e1; border-radius: 4px; }
::-webkit-scrollbar-thumb:hover { background: #94a3b8; }
```

  {% endfor %}

"""Visual smoke tests for dashboard pages."""
from pathlib import Path

import pytest
from fastapi.testclient import TestClient

from forge_agent.dashboard.app import create_app


def test_index_has_tailwind_classes(tmp_path: Path) -> None:

    assert "cdn.tailwindcss.com" in r.text

    # Custom CSS
    assert "/static/forge.css" in r.text

    client = TestClient(create_app(project_root=tmp_path))
    r = client.get("/")
    assert r.status_code == 200
    assert "status-pill" in r.text
